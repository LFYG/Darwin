# Make iOS Toolchains
mkdir $HOME/Developer/CommandLineTools

cp -r /Volumes/Xcode/Xcode.app/Contents/Developer/usr $HOME/Developer/CommandLineTools/
cp -r /Volumes/Xcode/Xcode.app/Contents/Developer/Toolchains/XcodeDefault.xctoolchain/usr $HOME/Developer/CommandLineTools/
cp -r /Volumes/Xcode/Xcode.app/Contents/Developer/Platforms/iPhoneOS.platform/Developer/usr $HOME/Developer/CommandLineTools/

cd $HOME/Developer/CommandLineTools
ln -s ../SDKs SDKs

# backup
cd ../../
tar Jcvf iOS-Toolchains.tar.xz Developer

# Switch iOS Toolchains
sudo xcode-select --switch $HOME/Developer/CommandLineTools
