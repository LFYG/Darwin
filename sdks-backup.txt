壓縮 SDKs
open /Volumes/Xcode/Xcode.app/Contents/Developer/Platforms/iPhoneOS.platform/Developer/SDKs
open /Volumes/Xcode/Xcode.app/Contents/Developer/Platforms/MacOSX.platform/Developer/SDKs

mkdir -p $HOME/Developer/SDKs
open $HOME/Developer/SDKs

cp sdk to $HOME/Developer/SDKs/

清除 .DS_Store
cd $HOME/Developer
DS=$(/usr/bin/find * | /usr/bin/grep '.DS_Store')
for i in $DS; do
    /bin/rm "$i"
done

SDK="iPhoneOS8.1.sdk"
SDK="MacOSX10.11.sdk"
cd $HOME/Developer/SDKs
tar Jcvf /Volumes/APP/SDKs/"$SDK".tar.xz "$SDK"/
