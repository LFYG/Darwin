壓縮 SDKs
open /Volumes/Xcode/Xcode.app/Contents/Developer/Platforms/iPhoneOS.platform/Developer/SDKs
open /Volumes/Xcode/Xcode.app/Contents/Developer/Platforms/MacOSX.platform/Developer/SDKs

mkdir $HOME/SDKs
open $HOME/SDKs

cp sdk to $HOME/SDKs/

清除 .DS_Store
cd $HOME/SDKs
DS=$(/usr/bin/find * | /usr/bin/grep '.DS_Store')
for i in $DS; do
    /bin/rm "$i"
done

SDK="iPhoneOS8.1.sdk"
SDK="MacOSX10.11.sdk"

tar Jcvf /Volumes/DATA/SDKs/"$SDK".tar.xz "$SDK"/

解壓縮 SDKs

SDK="iPhoneOS6.1.sdk"
SDK="MacOSX10.8.sdk"
